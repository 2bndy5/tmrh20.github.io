71 void RF24Network::begin(uint8_t _channel, uint16_t _node_address )
72 {
73  if (! is_valid_address(_node_address) )
74  return;
75 
76  node_address = _node_address;
77 
78  if ( ! radio.isValid() ){
79  return;
80  }
81 
82  //radio.stopListening();
83  // Set up the radio the way we want it to look
84  radio.setChannel(_channel);
85  radio.enableDynamicAck();
86  radio.enableDynamicPayloads();
87 
88  // Use different retry periods to reduce data collisions
89  uint8_t retryVar = (((node_address % 6)+1) *2) + 3;
90  radio.setRetries(retryVar, 5);
91  txTimeout = 25;
92  routeTimeout = txTimeout*9; // Adjust for max delay per node
93 
94 
95 #if defined (DUAL_HEAD_RADIO)
96  radio1.setChannel(_channel);
97  radio1.enableDynamicAck();
98  radio1.enableDynamicPayloads();
99 #endif
100 
101  // Setup our address helper cache
102  setup_address();
103 
104  // Open up all listening pipes
105  uint8_t i = 6;
106  while (i--){
107  radio.openReadingPipe(i,pipe_address(_node_address,i));
108  }
109 
110  radio.startListening();
111 
112 }

333 bool RF24Network::appendFragmentToFrame(RF24NetworkFrame frame) {
334 
335  // This is the first of 2 or more fragments.
336  if (frame.header.type == NETWORK_FIRST_FRAGMENT){
337  if( frameFragmentsCache.count(std::make_pair(frame.header.id,frame.header.from_node)) == 0 ){
338 
339  if(frame.header.reserved > (MAX_PAYLOAD_SIZE /24) + 1 ){
340  IF_SERIAL_DEBUG_FRAGMENTATION( printf("%u FRG Too many fragments in payload %u, dropping...",millis(),frame.header.reserved); );
341  // If there are more fragments than we can possibly handle, return
342  return false;
343  }
344  // Search for duplicate frames in the cache with an old header id, delete the older one
345  for (std::map<std::pair<uint16_t, uint16_t>, RF24NetworkFrame>::iterator it=frameFragmentsCache.begin(); it!=frameFragmentsCache.end(); ++it){
346  if(it->first.second == frame.header.from_node && it->first.first < frame.header.id ){
347  IF_SERIAL_DEBUG_FRAGMENTATION( printf("%u FRG Erased %d from 0%o\n",millis(),it->first.first,it->first.second); );
348  frameFragmentsCache.erase(it);
349  break;
350  }
351  }
352  frameFragmentsCache[ std::make_pair(frame.header.id,frame.header.from_node) ] = frame;
353  return true;
354  }
355  return false;
356  }else
357 
358  if ( frame.header.type == NETWORK_MORE_FRAGMENTS || frame.header.type == NETWORK_MORE_FRAGMENTS_NACK ){
359 
360  if( frameFragmentsCache.count(std::make_pair(frame.header.id,frame.header.from_node)) < 1 ){
361  return false;
362  }
363  RF24NetworkFrame *f = &(frameFragmentsCache[ std::make_pair(frame.header.id,frame.header.from_node) ]);
364  if( f->header.reserved - 1 == frame.header.reserved ){
365  // Cache the fragment
366  memcpy(f->message_buffer+f->message_size, frame.message_buffer, frame.message_size);
367  f->message_size += frame.message_size; //Increment message size
368  f->header = frame.header; //Update header
369  return true;
370 
371  } else {
372  IF_SERIAL_DEBUG_FRAGMENTATION(printf("%u: FRG Dropping fragment for frame with header id:%d, out of order fragment(s).\n",millis(),frame.header.id););
373  return false;
374  }
375 
376  }else
377  if ( frame.header.type == NETWORK_LAST_FRAGMENT ){
378 
379  //We have received the last fragment
380  if(frameFragmentsCache.count(std::make_pair(frame.header.id,frame.header.from_node)) < 1){
381  return false;
382  }
383  //Create pointer to the cached frame
384  RF24NetworkFrame *f = &(frameFragmentsCache[ std::make_pair(frame.header.id,frame.header.from_node) ]);
385 
386  if( f->message_size + frame.message_size > MAX_PAYLOAD_SIZE){
387  IF_SERIAL_DEBUG_FRAGMENTATION( printf("%u FRG Frame of size %u plus enqueued frame of size %u exceeds max payload size \n",millis(),frame.message_size,f->message_size); );
388  return false;
389  }
390  //Error checking for missed fragments and payload size
391  if ( f->header.reserved-1 != 1 ) {
392  IF_SERIAL_DEBUG_FRAGMENTATION(printf("%u: FRG Duplicate or out of sequence frame %d, expected %d. Cleared.\n",millis(),frame.header.reserved,f->header.reserved););
393  //frameFragmentsCache.erase( std::make_pair(frame.header.id,frame.header.from_node) );
394  return false;
395  }
396  //The user specified header.type is sent with the last fragment in the reserved field
397  frame.header.type = frame.header.reserved;
398  frame.header.reserved = 1;
399 
400  //Append the received fragment to the cached frame
401  memcpy(f->message_buffer+f->message_size, frame.message_buffer, frame.message_size);
402  f->message_size += frame.message_size; //Increment message size
403  f->header = frame.header; //Update header
404  return true;
405  }
406  return false;
407 }
